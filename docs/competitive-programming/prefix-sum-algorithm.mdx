## Introduction

You are given an array of `N` elements. It's Prefix Sum is another array of same size.
where PrefixSum[i] is sum of all prefix elements of the given array. Formally, 
PrefixSum[i] = ar[0] `+` ar[1] `+` ar[2]... ar[i].


## Explanation

Let us have one example:

```
Let us say, given array is ar[] = {1,2,3,4};
then it's prefix sum array will be prefix_sum_array[] = {1,3,6,10};
here, prefix_sum_array[1] = ar[0] `+` ar[1].
thus will goes on for the rest indexes of Prefix Sum array.
```

## Algorithm

 - New Prefix Sum array's first element will be same as given array's first element.
 - Next we will run a loop from index `1` to `last` index.
 - In Each iteration, Prefix Sum array's current element will be sum of previous element of this array and current index value of our given array.
 - So, iterations will be like this, prefix_sum_array[1] = prefix_sum_array[0] `+` ar[1].
 - prefix_sum_array[2] = prefix_sum_array[1] `+` ar[2] and so on.

```cpp
#include <bits/stdc++.h>
using namespace std;

// this function will give us the new Prefix Sum array
void prefix_sum_array(int ar[] ,int N)
{
    int pref_ar[N+1] = {0};
    pref_ar[0] = ar[0];
    /*we will add the current element of ar[] array 
    and previous element of pref_ar [] array */
    for(int i = 1; i < N; i++)
    {
        pref_ar[i] = ar[i] + pref_ar[i - 1];
    }
    cout << "The New Prefix sum array is \n";
    for(int i = 0; i < N; i++)
    {
        cout << pref_ar[i] << ' ';
    }
    cout << endl;
    return;
}

int main()
{
    cout << "Enter the size of array \n";
    int N;
    cin >> N;
    int ar[N + 1];
    cout << "Enter array elements \n";
    for(int i = 0; i < N; i++)
    {
        cin >> ar[i];
    }
    prefix_sum_array(ar , N);
}
```

```java
import java.util.Scanner;
import java.lang.*;

public class PrefixSum 
{
    // this function will give us prefix sum of the array
	static void prefix_sum_array(int ar[], int N)
	{
	    int pref_ar[] = new int[N];

	    pref_ar[0] = ar[0];
	    /* we will add the current element of ar[] array
         and previous element of pref_ar [] array */
        for(int i = 1; i < N; i++)
        {
           pref_ar[i] = ar[i] + pref_ar[i - 1];
        }
        System.out.println("The New Prefix sum array is \n");
		for (int i = 0; i < N; i++)
        {
            System.out.print(pref_ar[i] + " ");
        }
        System.out.println('\n');
	}

	public static void main(String args[]) 
	{
		Scanner scan = new Scanner(System.in);
		System.out.print("Enter the size of array \n");
		int number = scan.nextInt();

		int[] ar = new int[number];
		System.out.println("Enter array elements \n"); 
		for(int i = 0; i < number; i++)
		{
		    ar[i] = scan.nextInt();
		}
		scan.close();
		prefix_sum_array(ar,number);

		}
}
```
